// Convert text to speech with just a few lines async function convertToSpeech() { const response = await fetch('https://api.speaksynth.com/api/v1/synthesize', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-API-Key': 'YOUR_API_KEY' }, body: JSON.stringify({ text: "Hello, this is a premium voice!", format: 1 // 1 = WAV, 2 = OPUS }) });   const audioBlob = await response.blob(); const audioUrl = URL.createObjectURL(audioBlob); const audio = new Audio(audioUrl); audio.play(); }
